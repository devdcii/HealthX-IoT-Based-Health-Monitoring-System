void initWeightScale() {
  Serial.print("Initializing Weight Scale (HX711)... ");
  weightScale.begin(WEIGHT_DOUT_PIN, WEIGHT_SCK_PIN);
  delay(1000);
  
  if (weightScale.is_ready()) {
    weightScale.set_scale(weight_calibration_factor);
    weightScale.tare(20);
    weightSensorReady = true;
    Serial.println(" ✅ SUCCESS");
  } else {
    Serial.println(" ❌ FAILED");

}

void updateWeight() {
  if (weightScale.is_ready()) {
    float weight_grams = weightScale.get_units(5);
    current_weight = weight_grams / 1000.0;
    if (current_weight < 0) current_weight = abs(current_weight);
    if (current_weight < 0.1) current_weight = 0.0;
  }
}

void initHeartRateSensor() {
  Serial.print("Initializing Heart Rate & SpO2 (MAX30102)... ");
  
  if (!particleSensor.begin(Wire, I2C_SPEED_FAST)) {
    Serial.println("❌ FAILED");
    hrSensorReady = false;
    return;
  }
  
  byte ledBrightness = 50;
  byte sampleAverage = 1;
  byte ledMode = 2;
  byte sampleRate = 100;
  int pulseWidth = 69;
  int adcRange = 4096;
  
  particleSensor.setup(ledBrightness, sampleAverage, ledMode, sampleRate, pulseWidth, adcRange);
  
  for (int i = 0; i < RATE_SIZE; i++) {
    rates[i] = 0;
  }
  
  hrSensorReady = true;
  hrInitialCollectionDone = false;
  hrSampleIndex = 0;
  
  Serial.println("✅ SUCCESS (Live readings mode)");
}

void updateHeartRateAndSpO2() {
  if (!hrSensorReady) return;
  
  // Always check sensor and handle requests
  server.handleClient();
  particleSensor.check();
  
  // Initial collection - collect multiple samples but yield to server
  if (!hrInitialCollectionDone) {
    while (hrSampleIndex < 100) {
      // Handle server requests every 10 samples
      if (hrSampleIndex % 10 == 0) {
        server.handleClient();
      }
      
      while (particleSensor.available() == false) {
        particleSensor.check();
        server.handleClient(); // Keep server responsive
      }
      
      redBuffer[hrSampleIndex] = particleSensor.getRed();
      irBuffer[hrSampleIndex] = particleSensor.getIR();
      particleSensor.nextSample();
      hrSampleIndex++;
    }
    
    maxim_heart_rate_and_oxygen_saturation(irBuffer, 100, redBuffer, &spo2, &validSPO2, &heartRate, &validHeartRate);
    hrInitialCollectionDone = true;
    hrSampleIndex = 0;
    return;
  }
  
  // Shift buffer
  for (byte i = 25; i < 100; i++) {
    redBuffer[i - 25] = redBuffer[i];
    irBuffer[i - 25] = irBuffer[i];
  }
  
  // Collect 25 new samples with server handling
  for (byte i = 75; i < 100; i++) {
    // Handle server every 5 samples
    if (i % 5 == 0) {
      server.handleClient();
    }
    
    while (particleSensor.available() == false) {
      particleSensor.check();
      server.handleClient(); // Keep responsive
    }
    
    redBuffer[i] = particleSensor.getRed();
    irBuffer[i] = particleSensor.getIR();
    particleSensor.nextSample();
    
    long irValue = irBuffer[i];
    
    if (checkForBeat(irValue) == true) {
      long delta = millis() - lastBeat;
      lastBeat = millis();
      
      beatsPerMinute = 60 / (delta / 1000.0);
      beatAvg = (beatAvg + beatsPerMinute) / 2;
    }
    
    if (millis() - lastBeat > 10000) {
      beatsPerMinute = 0;
      beatAvg = (beatAvg + beatsPerMinute) / 2;
    }
  }
  
  maxim_heart_rate_and_oxygen_saturation(irBuffer, 100, redBuffer, &spo2, &validSPO2, &heartRate, &validHeartRate);
}

  }
}

long readHX710B() {
  unsigned long timeout = micros();
  while (digitalRead(BP_HX_DOUT_PIN) == HIGH) {
    if (micros() - timeout > 100000) return 0;
  }
  
  long value = 0;
  for (int i = 0; i < 24; i++) {
    digitalWrite(BP_HX_SCK_PIN, HIGH);
    delayMicroseconds(1);
    value = (value << 1) | digitalRead(BP_HX_DOUT_PIN);
    digitalWrite(BP_HX_SCK_PIN, LOW);
    delayMicroseconds(1);
  }
  
  digitalWrite(BP_HX_SCK_PIN, HIGH);
  delayMicroseconds(1);
  digitalWrite(BP_HX_SCK_PIN, LOW);
  
  if (value & 0x800000) value |= 0xFF000000;
  
  return value;
}

float readBPPressure() {
  static float filtered = 0.0;
  const float ALPHA = 0.3;
  
  long sum = 0;
  int valid = 0;
  
  for (int i = 0; i < 3; i++) {
    long raw = readHX710B();
    if (raw != 0) {
      sum += raw;
      valid++;
    }
    delay(5);
  }
  
  if (valid == 0) return filtered;
  
  float rawAvg = sum / (float)valid;
  float newPressure = (rawAvg - bp_zero_offset) / bp_scale_factor;
  
  filtered = ALPHA * newPressure + (1 - ALPHA) * filtered;
  
  if (filtered < 0) filtered = 0;
  if (filtered > 300) filtered = 300;
  
  return filtered;
}

void setPumpPower(int dutyCycle) {
  if (!pumpPWMConfigured) {
    ledcAttach(PUMP_PIN, 1000, 8);
    pumpPWMConfigured = true;
  }
  ledcWrite(PUMP_PIN, dutyCycle);
}

void stopPump() {
  if (pumpPWMConfigured) {
    ledcWrite(PUMP_PIN, 0);
  } else {
    digitalWrite(PUMP_PIN, LOW);
  }
}

void handleHealthData() {
  updateWeight();
  
  if (heightSensorReady) {
    heightSensor.read();
    if (!heightSensor.timeoutOccurred()) {
      float dist = heightSensor.ranging_data.range_mm / 10.0;
      height_cm = sensor_mount_height_cm - dist;
    }
  }
  
  if (current_weight > 0 && height_cm > 0) {
    bmi = current_weight / ((height_cm / 100.0) * (height_cm / 100.0));
  }
  
  DynamicJsonDocument doc(2048);
  
  doc["weight"]["value"] = current_weight;
  doc["weight"]["unit"] = "kg";
  doc["weight"]["status"] = weightSensorReady;
  
  doc["bmi"]["value"] = bmi;
  doc["bmi"]["height"] = height_cm;
  doc["bmi"]["unit"] = "";
  doc["bmi"]["status"] = (weightSensorReady && heightSensorReady);
  
  doc["heartRate"]["value"] = beatAvg;
  doc["heartRate"]["unit"] = "bpm";
  doc["heartRate"]["status"] = hrSensorReady;
  
  doc["spo2"]["value"] = spo2;
  doc["spo2"]["unit"] = "%";
  doc["spo2"]["status"] = hrSensorReady;
  
  doc["temperature"]["value"] = bodyTemperature;
  doc["temperature"]["unit"] = "°C";
  doc["temperature"]["status"] = tempSensorReady;
  
  doc["bloodPressure"]["systolic"] = systolic;
  doc["bloodPressure"]["diastolic"] = diastolic;
  doc["bloodPressure"]["unit"] = "mmHg";
  doc["bloodPressure"]["status"] = bpSensorReady;
  
  doc["timestamp"] = millis();
  doc["deviceIP"] = WiFi.localIP().toString();
  
  String response;
  serializeJson(doc, response);
  server.send(200, "application/json", response);
}

void loop() {
  // PRIORITY 1: Always handle server requests first
  server.handleClient();
  
  // PRIORITY 2: Check WiFi
  if (WiFi.status() != WL_CONNECTED) {
    Serial.println("⚠️  WiFi disconnected! Reconnecting...");
    setupWiFi();
  }
  
  unsigned long currentTime = millis();
  
  // Update weight
  if (currentTime - last_weight_update >= WEIGHT_UPDATE_INTERVAL) {
    if (weightSensorReady) {
      updateWeight();
    }
    last_weight_update = currentTime;
  }
  
  // Update temperature
  if (currentTime - last_temp_update >= TEMP_UPDATE_INTERVAL) {
    if (tempSensorReady) {
      updateTemperature();
    }
    last_temp_update = currentTime;
  }
  
  // Update HR & SpO2 (with embedded server.handleClient() calls)
  if (hrSensorReady) {
    updateHeartRateAndSpO2();
  }
  
  // Small delay to prevent watchdog
  delay(1);
}

class ApiService {
  // ===== AUTHENTICATION =====

  static Future<Map<String, dynamic>> login(String email, String password) async {
    try {
      final response = await http.post(
        Uri.parse(ApiConfig.authEndpoint),
        headers: {'Content-Type': 'application/json'},
        body: jsonEncode({
          'action': 'login',
          'email': email,
          'password': password,
        }),
      );

      return jsonDecode(response.body);
    } catch (e) {
      return {'success': false, 'message': 'Connection error: $e'};
    }
  }

  static Future<Map<String, dynamic>> signup(
      String name, String email, String password) async {
    try {
      final response = await http.post(
        Uri.parse(ApiConfig.authEndpoint),
        headers: {'Content-Type': 'application/json'},
        body: jsonEncode({
          'action': 'signup',
          'name': name,
          'email': email,
          'password': password,
        }),
      );

      return jsonDecode(response.body);
    } catch (e) {
      return {'success': false, 'message': 'Connection error: $e'};
    }
  }

  // ===== PATIENT MANAGEMENT =====

  static Future<Map<String, dynamic>> getPatients() async {
    try {
      final response = await http.get(
        Uri.parse(ApiConfig.getPatientsEndpoint),
      );

      if (response.statusCode == 200) {
        return jsonDecode(response.body);
      }
      return {'success': false, 'message': 'Failed to fetch patients'};
    } catch (e) {
      return {'success': false, 'message': 'Connection error: $e'};
    }
  }

  static Future<Map<String, dynamic>> deletePatient(dynamic patientId) async {
    try {
      int id = patientId is String ? int.parse(patientId) : patientId;

      final response = await http.post(
        Uri.parse(ApiConfig.deletePatientEndpoint),
        headers: {'Content-Type': 'application/json'},
        body: jsonEncode({'patient_id': id}),
      );

      if (response.statusCode == 200) {
        return jsonDecode(response.body);
      }
      return {'success': false, 'message': 'Server error: ${response.statusCode}'};
    } catch (e) {
      return {'success': false, 'message': 'Connection error: $e'};
    }
  }

  // ===== HEALTH READINGS =====

  static Future<Map<String, dynamic>> getReadings(String userEmail) async {
    try {
      final response = await http.get(
        Uri.parse('${ApiConfig.getReadingsEndpoint}?user_email=$userEmail'),
      );

      if (response.statusCode == 200) {
        return jsonDecode(response.body);
      }
      return {'success': false, 'message': 'Failed to fetch readings'};
    } catch (e) {
      return {'success': false, 'message': 'Connection error: $e'};
    }
  }

  static Future<Map<String, dynamic>> saveReading(
      Map<String, dynamic> readingData) async {
    try {
      final response = await http.post(
        Uri.parse(ApiConfig.saveReadingEndpoint),
        headers: {'Content-Type': 'application/json'},
        body: jsonEncode(readingData),
      );

      if (response.statusCode == 200) {
        return jsonDecode(response.body);
      }
      return {'success': false, 'message': 'Failed to save reading'};
    } catch (e) {
      return {'success': false, 'message': 'Connection error: $e'};
    }
  }

  static Future<Map<String, dynamic>> deleteReading(dynamic readingId) async {
    try {
      int id = readingId is String ? int.parse(readingId) : readingId;

      final response = await http.post(
        Uri.parse(ApiConfig.deleteReadingEndpoint),
        headers: {'Content-Type': 'application/json'},
        body: jsonEncode({'reading_id': id}),
      );

      if (response.statusCode == 200) {
        return jsonDecode(response.body);
      }
      return {'success': false, 'message': 'Failed to delete reading'};
    } catch (e) {
      return {'success': false, 'message': 'Connection error: $e'};
    }
  }

  static Future<Map<String, dynamic>> updateReading(
      dynamic readingId, Map<String, dynamic> readingData) async {
    try {
      int id = readingId is String ? int.parse(readingId) : readingId;

      final response = await http.post(
        Uri.parse(ApiConfig.updateReadingEndpoint),
        headers: {'Content-Type': 'application/json'},
        body: jsonEncode({
          'reading_id': id,
          'bmi': readingData['bmi'],
          'heart_rate': readingData['heart_rate'],
          'spo2': readingData['spo2'],
          'temperature': readingData['temperature'],
          'systolic': readingData['systolic'],
          'diastolic': readingData['diastolic'],
        }),
      );

      if (response.statusCode == 200) {
        return jsonDecode(response.body);
      }
      return {'success': false, 'message': 'Failed to update reading'};
    } catch (e) {
      return {'success': false, 'message': 'Connection error: $e'};
    }
  }

  // ===== ESP32 SENSOR DATA =====

  /// Get all health data from ESP32 (combined endpoint)
  static Future<Map<String, dynamic>> getEsp32HealthData() async {
    try {
      final response = await http.get(
        Uri.parse(ApiConfig.esp32HealthEndpoint),
      ).timeout(Duration(seconds: 5));

      if (response.statusCode == 200) {
        final data = jsonDecode(response.body);
        return {'success': true, ...data};
      }
      return {'success': false, 'message': 'Failed to fetch ESP32 data'};
    } catch (e) {
      return {'success': false, 'message': 'ESP32 connection error: $e'};
    }
  }

        Uri.parse('${ApiConfig.esp32Url}/tare'),
        headers: {'Content-Type': 'application/json'},

      isStarted = true;

      finalData = {};
    });
    await _activateSensor();
    _startRealtimePolling();
  }

  Future<void> _activateSensor() async {
    await Future.delayed(Duration(milliseconds: 1500));
  }

  void _startRealtimePolling() {
    _sensorTimer = Timer.periodic(Duration(milliseconds: 500), (timer) async {
      final result = await ApiService.getEsp32HealthData();
      if (result['success'] != false && mounted) {
        setState(() {
          isLoading = false;
          _updateLiveData(result);
        });
      }
    });
  }

  void _updateLiveData(Map<String, dynamic> espData) {
    switch (widget.paramKey) {
      case 'bmi':
        liveData = {
          'weight': espData['weight']?['value']?.toDouble() ?? 0.0,
          'height': espData['bmi']?['height']?.toDouble() ?? 170.0,
          'bmi': espData['bmi']?['value']?.toDouble() ?? 0.0,
        };
        break;
      case 'temperature':
        liveData = {
          'temperature': espData['temperature']?['value']?.toDouble() ?? 0.0,
        };
        break;
      case 'vitals':
        liveData = {
          'heart_rate': espData['heartRate']?['value'] ?? 0,
          'spo2': espData['spo2']?['value'] ?? 0,
        };
        break;
      case 'bp':
        liveData = {
          'systolic': espData['bloodPressure']?['systolic'] ?? 0,
          'diastolic': espData['bloodPressure']?['diastolic'] ?? 0,
        };
        break;
    }
  }

  void _resetMeasurement() {
    setState(() {
      isStarted = false;
      isLoading = false;
      isSaved = false;
      liveData = {};
      finalData = {};
    });
    _sensorTimer?.cancel();
  }
