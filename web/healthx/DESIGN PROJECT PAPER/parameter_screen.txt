import 'package:flutter/material.dart';
import 'dart:async';
import 'healthworker_dashboard.dart';
import 'api_service.dart';

class ParameterMeasurementScreen extends StatefulWidget {
  final String title;
  final IconData icon;
  final Color color;
  final String paramKey;
  final String patientName;
  final Function(Map<String, dynamic>) onDataSaved;

  const ParameterMeasurementScreen({
    Key? key,
    required this.title,
    required this.icon,
    required this.color,
    required this.paramKey,
    required this.patientName,
    required this.onDataSaved,
  }) : super(key: key);

  @override
  State<ParameterMeasurementScreen> createState() =>
      _ParameterMeasurementScreenState();
}

class _ParameterMeasurementScreenState
    extends State<ParameterMeasurementScreen> {
  bool isStarted = false;
  bool isLoading = false;
  bool isSaved = false;
  bool showInstructions = true;
  Timer? _sensorTimer;
  Map<String, dynamic> liveData = {};
  Map<String, dynamic> finalData = {};

  Map<String, String> parameterImages = {
    'bmi': 'images/body-mass-index.png',
    'height': 'images/height.png',
    'weight': 'images/weight.png',
    'temperature': 'images/body-temperature.png',
    'heart_rate': 'images/oxygen-saturation.png',
    'spo2': 'images/oxygen-saturation.png',
    'bp': 'images/blood-pressure.png',
    'vitals': 'images/oxygen-saturation.png',
  };

  @override
  void initState() {
    super.initState();
    WidgetsBinding.instance.addPostFrameCallback((_) {
      _showInstructionsDialog();
    });
  }

  @override
  void dispose() {
    _sensorTimer?.cancel();
    super.dispose();
  }

  void _showInstructionsDialog() {
    Navigator.of(context).push(
      MaterialPageRoute(
        fullscreenDialog: true,
        builder: (context) => _InstructionsDialog(
          paramKey: widget.paramKey,
          title: widget.title,
          color: widget.color,
          imagePath: parameterImages[widget.paramKey] ?? 'images/oxygen-saturation.png',
          onComplete: () {
            Navigator.pop(context);
            setState(() => showInstructions = false);
          },
        ),
      ),
    );
  }

  void _startMeasurement() async {
    setState(() {
      isStarted = true;
      isLoading = true;
      isSaved = false;
      liveData = {};
      finalData = {};
    });
    await _activateSensor();
    _startRealtimePolling();
  }

  Future<void> _activateSensor() async {
    await Future.delayed(Duration(milliseconds: 1500));
  }

  void _startRealtimePolling() {
    _sensorTimer = Timer.periodic(Duration(milliseconds: 500), (timer) async {
      final result = await ApiService.getEsp32HealthData();
      if (result['success'] != false && mounted) {
        setState(() {
          isLoading = false;
          _updateLiveData(result);
        });
      }
    });
  }

  void _updateLiveData(Map<String, dynamic> espData) {
    print('Updating live data for ${widget.paramKey}');
    print('ESP Data received: $espData');

    switch (widget.paramKey) {
      case 'bmi':
        liveData = {
          'weight': espData['weight']?['value']?.toDouble() ?? 0.0,
          'height': espData['bmi']?['height']?.toDouble() ?? 170.0,
          'bmi': espData['bmi']?['value']?.toDouble() ?? 0.0,
        };
        print('BMI Live Data: $liveData');
        break;

      case 'temperature':
        double temp = espData['temperature']?['value']?.toDouble() ?? 0.0;
        liveData = {
          'temperature': temp,
        };
        print('Temperature Live Data: $liveData (from ESP: $temp)');
        break;

      case 'vitals':
        int hr = espData['heartRate']?['value'] ?? 0;
        int spo2Val = espData['spo2']?['value'] ?? 0;
        liveData = {
          'heart_rate': hr,
          'spo2': spo2Val,
        };
        print('Vitals Live Data: $liveData');
        break;

      case 'bp':
        int sys = espData['bloodPressure']?['systolic'] ?? 0;
        int dia = espData['bloodPressure']?['diastolic'] ?? 0;
        liveData = {
          'systolic': sys,
          'diastolic': dia,
        };
        print('BP Live Data: $liveData');
        break;
    }
  }

  void _resetMeasurement() {
    setState(() {
      isStarted = false;
      isLoading = false;
      isSaved = false;
      liveData = {};
      finalData = {};
    });
    _sensorTimer?.cancel();
  }

  Future<void> _tareScale() async {
    try {
      await ApiService.tareWeightScale();
      if (mounted) {
        ScaffoldMessenger.of(context).showSnackBar(
          SnackBar(
            content: Row(
              children: [
                Icon(Icons.check_circle_outline, color: AppColors.white),
                SizedBox(width: 12),
                Text('Scale tared successfully!'),
              ],
            ),
            backgroundColor: AppColors.success,
            behavior: SnackBarBehavior.floating,
            shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(12)),
          ),
        );
      }
    } catch (e) {
      if (mounted) {
        ScaffoldMessenger.of(context).showSnackBar(
          SnackBar(
            content: Row(
              children: [
                Icon(Icons.error_outline, color: AppColors.white),
                SizedBox(width: 12),
                Text('Failed to tare scale: $e'),
              ],
            ),
            backgroundColor: AppColors.error,
            behavior: SnackBarBehavior.floating,
            shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(12)),
          ),
        );
      }
    }
  }

  void _saveData() {
    if (liveData.isEmpty) return;
    setState(() {
      finalData = Map.from(liveData);
      isSaved = true;
    });
    widget.onDataSaved(finalData);

    if (mounted) {
      ScaffoldMessenger.of(context).showSnackBar(
        SnackBar(
          content: Row(
            children: [
              Icon(Icons.check_circle_outline, color: AppColors.white),
              SizedBox(width: 12),
              Text('${widget.title} data saved!'),
            ],
          ),
          backgroundColor: AppColors.success,
          behavior: SnackBarBehavior.floating,
          shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(12)),
        ),
      );
    }
  }

  String _getBMICategory(double bmi) {
    if (bmi < 18.5) return 'Underweight';
    if (bmi < 25) return 'Normal';
    if (bmi < 30) return 'Overweight';
    return 'Obese';
  }

  Color _getBMICategoryColor(double bmi) {
    if (bmi < 18.5) return Color(0xFF3B82F6);
    if (bmi < 25) return AppColors.success;
    if (bmi < 30) return Color(0xFFF59E0B);
    return AppColors.error;
  }

  String _getTemperatureStatus(double temp) {
    if (temp < 36.1) return 'Low';
    if (temp <= 37.2) return 'Normal';
    if (temp <= 38.0) return 'Elevated';
    return 'Fever';
  }

  Color _getTemperatureColor(double temp) {
    if (temp < 36.1) return Color(0xFF3B82F6);
    if (temp <= 37.2) return AppColors.success;
    if (temp <= 38.0) return Color(0xFFF59E0B);
    return AppColors.error;
  }

  String _getHeartRateStatus(int hr) {
    if (hr < 60) return 'Low';
    if (hr <= 100) return 'Normal';
    if (hr <= 120) return 'Elevated';
    return 'High';
  }

  Color _getHeartRateColor(int hr) {
    if (hr < 60) return Color(0xFF3B82F6);
    if (hr <= 100) return AppColors.success;
    if (hr <= 120) return Color(0xFFF59E0B);
    return AppColors.error;
  }

  String _getSpO2Status(int spo2) {
    if (spo2 >= 95) return 'Normal';
    if (spo2 >= 90) return 'Low';
    return 'Critical';
  }

  Color _getSpO2Color(int spo2) {
    if (spo2 >= 95) return AppColors.success;
    if (spo2 >= 90) return Color(0xFFF59E0B);
    return AppColors.error;
  }

  String _getBloodPressureStatus(int systolic, int diastolic) {
    if (systolic < 120 && diastolic < 80) return 'Normal';
    if (systolic < 130 && diastolic < 80) return 'Elevated';
    if (systolic < 140 || diastolic < 90) return 'Stage 1 Hypertension';
    if (systolic < 180 || diastolic < 120) return 'Stage 2 Hypertension';
    return 'Hypertensive Crisis';
  }

  Color _getBloodPressureColor(int systolic, int diastolic) {
    if (systolic < 120 && diastolic < 80) return AppColors.success;
    if (systolic < 130 && diastolic < 80) return Color(0xFF3B82F6);
    if (systolic < 140 || diastolic < 90) return Color(0xFFF59E0B);
    return AppColors.error;
  }

                widget.title,

      body: Column(
        children: [
          Expanded(
            child: SingleChildScrollView(

              child: Column(

                children: [
                  Container(
                    padding: EdgeInsets.all(16),

                        begin: Alignment.topLeft,

                      ),
                      borderRadius: BorderRadius.circular(16),
                      boxShadow: [
                        BoxShadow(
                          color: Color(0xFF1848A0).withOpacity(0.3),

                      ],

                      children: [

  List<Widget> _buildBMIDisplay() {
    double weight = liveData['weight'] ?? 0.0;
    double height = liveData['height'] ?? 0.0;
    double bmi = liveData['bmi'] ?? 0.0;

    return [
      _buildDataRow('Weight', weight.toStringAsFixed(1), 'kg', 'images/weight.png'),
      Container(
        margin: EdgeInsets.only(bottom: 16),
        child: SizedBox(
          width: double.infinity,
          height: 50,
          child: OutlinedButton(
            onPressed: _tareScale,
            style: OutlinedButton.styleFrom(
              side: BorderSide(color: Color(0xFF1848A0), width: 2),
              shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(12)),
            ),
            child: Row(
              mainAxisAlignment: MainAxisAlignment.center,
              children: [
                Icon(Icons.refresh, color: Color(0xFF1848A0), size: 20),
                SizedBox(width: 8),
                Text(
                  'Tare Scale',
                  style: TextStyle(
                    fontSize: 14,
                    fontWeight: FontWeight.bold,
                    color: Color(0xFF1848A0),
                  ),
                ),
              ],
            ),
          ),
        ),
      ),
      _buildDataRow('Height', height.toStringAsFixed(1), 'cm', 'images/height.png'),
      _buildDataRow('BMI', bmi.toStringAsFixed(1), '', 'images/body-mass-index.png'),
      if (bmi > 0) ...[
        SizedBox(height: 8),
        _buildStatusIndicator(
          'BMI Category',
          _getBMICategory(bmi),
          _getBMICategoryColor(bmi),
          Icons.analytics,
        ),
      ],
    ];
  }

  List<Widget> _buildTemperatureDisplay() {
    double temp = liveData['temperature'] ?? 0.0;

    return [
      _buildDataRow('Body Temperature', temp.toStringAsFixed(1), '°C', 'images/body-temperature.png'),
      if (temp > 0) ...[
        SizedBox(height: 8),
        _buildStatusIndicator(
          'Temperature Status',
          _getTemperatureStatus(temp),
          _getTemperatureColor(temp),
          Icons.thermostat,
        ),
      ],
    ];
  }

  List<Widget> _buildVitalsDisplay() {
    int hr = liveData['heart_rate'] ?? 0;
    int spo2 = liveData['spo2'] ?? 0;

    return [
      _buildDataRow('Heart Rate', hr.toString(), 'bpm', 'images/oxygen-saturation.png'),
      _buildDataRow('Blood Oxygen (SpO2)', spo2.toString(), '%', 'images/oxygen-saturation.png'),
      if (hr > 0 || spo2 > 0) ...[
        SizedBox(height: 8),
        if (hr > 0)
          Padding(
            padding: EdgeInsets.only(bottom: 8),
            child: _buildStatusIndicator(
              'Heart Rate Status',
              _getHeartRateStatus(hr),
              _getHeartRateColor(hr),
              Icons.favorite,
            ),
          ),
        if (spo2 > 0)
          _buildStatusIndicator(
            'Oxygen Status',
            _getSpO2Status(spo2),
            _getSpO2Color(spo2),
            Icons.air,
          ),
      ],
    ];
  }

  List<Widget> _buildBPDisplay() {
    int systolic = liveData['systolic'] ?? 0;
    int diastolic = liveData['diastolic'] ?? 0;

    return [
      _buildDataRow('Systolic Pressure', systolic.toString(), 'mmHg', 'images/blood-pressure.png'),
      _buildDataRow('Diastolic Pressure', diastolic.toString(), 'mmHg', 'images/blood-pressure.png'),
      if (systolic > 0 && diastolic > 0) ...[
        SizedBox(height: 8),
        _buildStatusIndicator(
          'Blood Pressure Status',
          _getBloodPressureStatus(systolic, diastolic),
          _getBloodPressureColor(systolic, diastolic),
          Icons.monitor_heart,
        ),
      ],
    ];
  }

  Widget _buildBottomActions() {
    return Container(
